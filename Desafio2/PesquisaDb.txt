SELECT Cp.Numero AS Conta_a_Paga,P.Nome AS Nome, Cp.DataVancimento AS Validade, Cp.DataProrrogacao AS Dia_de_Pagamento, Cp.Valor AS Valor FROM ContasAPagar AS Cp
INNER JOIN Pessoas AS P
ON P.Codigo == Cp.CodigoFornecedor

SELECT Cp.Numero AS Conta_Paga,P.Nome AS Nome, Cp.DataVancimento AS Validade, Cp.DataProrrogacao AS Dia_Pago, Cp.Valor AS Valor FROM ContasPagas AS Cp
INNER JOIN Pessoas AS P
ON P.Codigo == Cp.CodigoFornecedor
